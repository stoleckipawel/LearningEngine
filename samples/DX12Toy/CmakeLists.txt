# DX12Toy sample CMake configuration

# Add executable target
add_executable(DX12Toy
    src/main.cpp
)

# Include engine headers and pch
target_include_directories(DX12Toy PRIVATE
    ${CMAKE_SOURCE_DIR}/engine/include
    ${CMAKE_SOURCE_DIR}/engine/include/third_party
    ${CMAKE_SOURCE_DIR}/engine/pch
)

# Require C++20
target_compile_features(DX12Toy PRIVATE cxx_std_20)

# Link against engine static library
target_link_libraries(DX12Toy PRIVATE Engine)

# Set SAMPLES_PATH environment variable for samples
set(SAMPLES_PATH "${CMAKE_CURRENT_SOURCE_DIR}")
set(ENV{SAMPLES_PATH} "${SAMPLES_PATH}")
